# bpm+tgr compatch
# Adds TGR "government_*" IG ideologies to BPM interest groups,
# so BPM IGs keep their identity but still have stances for TGR-only lawgroups/laws.
#
# Note: ideology_government_* are defined by TGR and are not overridden by BPM.

	on_enable = {
		ig:ig_armed_forces = {
			add_ideology = ideology_government_armed_forces
		}
	}
}
