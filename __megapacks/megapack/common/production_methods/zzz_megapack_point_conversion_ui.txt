
# UI/UX patch: make PSC point-conversion PMs look like construction-sector PMs in UI.
# Reason: budget/overview uses AccessConstructionSpeeds which is tied to CONSTRUCTION_CAMP_BUILDING
# (PSC sets it to building_construction_regulator), so the UI will show regulator PMs.

REPLACE:pm_wood_point_conversion = {
	texture = "gfx/interface/icons/production_method_icons/wooden_buildings.dds"
	is_default = yes
	is_hidden_when_unavailable = yes

	country_modifiers = {
		workforce_scaled = {
			country_construction_add = 1
		}
	}

	building_modifiers = {
		workforce_scaled = {
			goods_input_wood_construction_add = 10
		}
		level_scaled = {
			building_employment_bureaucrats_add = 20
		}
	}
	unlocking_production_methods = {
		pm_wood_point_conversion
	}
}
