##Tech&Res
#Commonores
agassiz_can_find_commonores_vein_trigger = {
	any_scope_building = {
		is_building_type = building_commonores_mine
		level > 0
		NOT = {
			has_modifier = agassiz_building_production_mult_modifier
		}
	}
}
